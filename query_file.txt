INSERT INTO push_notif_results (TYPE, value, TIMESTAMP) SELECT phylum AS notif_type, count(*) AS value, UNIX_TIMESTAMP(NOW()) FROM m_table_count_push_notif_send WHERE kingdom='success' AND TIMESTAMP BETWEEN (UNIX_TIMESTAMP(NOW()) - 120) AND (UNIX_TIMESTAMP(NOW()) - 90) AND phylum != "" GROUP BY phylum
insert into count_user_load_results(os, appver, value, timestamp) select installos as os, phylum as appver, count(*) as value, UNIX_TIMESTAMP(NOW()) from m_table_count_user_load where kingdom like 'createNewUser%' and game_id=3 and timestamp between (UNIX_TIMESTAMP(NOW()) - 120) AND (UNIX_TIMESTAMP(NOW()) - 90) group by installos, phylum
